\begin{itemize}
  \item The first item

\begin{enumerate}
  \item The first item

  \item The third etc \ldots
\end{enumerate}

\subsection{Description}

\begin{description}
  \item[First] The first item
  \item[Second] The second item
  \item[Third] The third etc \ldots
\end{description}

\section{Nested Lists}
\begin{enumerate}
  \item The first item
  \begin{enumerate}
    \item Nested item 1
    \item Nested item 2
  \end{enumerate}
  \item The second item
  \item The third etc \ldots
\end{enumerate}

  \item The third etc \ldots
\end{itemize}

  \item The first item
  \item The second item
  \item The third etc \ldots
\end{dinglist}
